def create_rules_dict(n_stages):
    meaning_columns = ['Unnamed: 0', '11p', '11m', '12p', '12m', '13p', '13m', '21p', '21m', '22p',
       '22m', '23p', '23m', '31p', '31m', '32p', '32m', '33p', '33m', '41p',
       '41m', '42p', '42m', '43p', '43m', '51p', '51m', '52p', '52m', '53p',
       '53m', '61p', '61m', '62p', '62m', '63p', '63m', '71p', '71m', '72p',
       '72m', '73p', '73m', '81p', '81m', '82p', '82m', '83p', '83m', '91p',
       '91m', '92p', '92m', '93p', '93m']
    df = pd.read_excel("Таблица решений.xlsx")[meaning_columns]
    df = df.rename(columns = {'Unnamed: 0': 'past'})

    past = df.iloc[:,0].astype(str).tolist()
    now = df.columns[1:]

    df = df.set_index('past')

    past_val = []
    now_val = []
    res_val = []
    for i in range(len(past)):
        for n in now:
            val = df.iloc[i][n] 
            isna = df.iloc[i].isna()[n] 
            if ~isna:
                res_val.append(str(int(val)))
                past_val.append(past[i])
                now_val.append(n)
    dfr = pd.DataFrame({'past':past_val,'now':now_val,'res':res_val})
    rules_dict = {(past_val[i],now_val[i]):res_val[i] for i in range(len(past_val))}
    
    for key in rules_dict.keys():
        if int(rules_dict[key][0]) > n_stages:
            rules_dict[key] = '1'

    return rules_dict

def generate_main_df(dbt,dhw, n_units = 50):
    dhw = dhw[(dhw['Результаты'].astype(int)>=1) & (dhw['Результаты'].astype(int)<101)].reset_index(drop = True)
    
    dhw['ID номера'] = dhw['ID номера'].str.split('\n').str[0:-1]
    dhw['Баллы за ответ ученика'] = dhw['Баллы за ответ ученика'].str.split('\n').str[0:-1]
    dhw['Максимальное кол-во баллов за данную задачу'] = dhw['Максимальное кол-во баллов за данную задачу'].str.split('\n').str[0:-1]
    
    rdf = pd.DataFrame({'id':dhw.explode('ID номера')['ID'], 'task_id':dhw.explode('ID номера')['ID номера'], 'ball':dhw.explode('Баллы за ответ ученика')['Баллы за ответ ученика'], 'mball':dhw.explode('Максимальное кол-во баллов за данную задачу')['Максимальное кол-во баллов за данную задачу']})
    rdf = rdf.dropna()
    rdf['isp'] = (rdf['mball'] == rdf['ball']).astype(int)

    rdf['ball'] = rdf['ball'].astype(int)
    rdf['mball'] = rdf['mball'].astype(int)

    mdf = rdf.groupby('task_id').agg(ball_sum = pd.NamedAgg('ball',aggfunc=sum),mball_sum = pd.NamedAgg('mball',aggfunc=sum), count = pd.NamedAgg('id',aggfunc='count'), isp = pd.NamedAgg('isp', aggfunc=sum)).reset_index()
    mdf = mdf[(mdf['mball_sum'] != 0) & (mdf['mball_sum'] >= mdf['ball_sum'])]
    mdf['av'] = mdf.apply(lambda x: x['ball_sum']/x['mball_sum'], axis = 1)
    avr = mdf['av'].mean()
    mdf['wr'] = mdf.apply(lambda x: 1 - ((x['count']/(x['count']+n_units))*x['av']+(n_units/(x['count']+n_units))*avr), axis = 1)
    mdf = mdf[mdf['count']>n_units]

    dbt['ID'] = dbt['ID'].astype(str)
    df = pd.merge(dbt,mdf,left_on='ID',right_on='task_id').drop('task_id',axis = 1)
    df = df.rename(columns = {'Номер в КИМ (порядок)':'kim','Тема':'theme'})
    df['kim'] = df['kim'].astype(int)
    return df

def classificator(umit_list):
    vectorizer = TfidfVectorizer(stop_words='english')
    X = vectorizer.fit_transform(umit_list)
    model = KMeans(n_clusters=3, init='k-means++', max_iter=100, n_init=1)
    model.fit(X)
    
    return vectorizer, model

def generate_task_state(taken_themes,kim,df):

    task_state = {}
    
    for theme in taken_themes:
        for level in range(1,4):
            task_id,task_dif = df[df['theme'] == theme].query(f'kim == {kim}').sample(1).iloc[0][['ID','wr']]
            task_state.update({str(taken_themes.index(theme)+1)+str(level):{'theme':theme, 'id':task_id, 'dif':round(task_dif,2)}})
    return task_state

def plot_bubbles(N,task_state,solved_tasks,correct_tasks,cp):    
    
    fig, ax = plt.subplots()

    grid = np.array(np.meshgrid(np.arange(1,4,1), -np.arange(1,N+1,1))).reshape(2,-1).T
    task_names = list(task_state.keys())

    solved = [1 if item in solved_tasks else 0 for item in task_names]
    correct = [1 if item in correct_tasks else 0 for item in task_names]

    #colors = np.kron(np.kron(np.arange(1,6).reshape(5,1),np.ones((1,3)))/6,np.ones((3,1)))
    colors = np.kron(np.random.rand(N,3), np.ones((3,1)))
    for i in range(3*N):
        if solved[i] == 1 and correct[i] == 1:
            ax.add_patch(mpl.patches.Circle(grid[i], radius=0.3, edgecolor="black", linewidth=1, facecolor = colors[i], alpha = 0.7))
        elif solved[i] == 1 and correct[i] == 0:
            ax.add_patch(mpl.patches.Circle(grid[i], radius=0.3, edgecolor=None, linewidth=None, facecolor = colors[i], alpha = 0.25))
        else:
            ax.add_patch(mpl.patches.Circle(grid[i], radius=0.3, edgecolor=None, alpha = 0.1))
        label(grid[i]+[0,0.5], f"{task_names[i][0]}.{task_names[i][1]}")


    for i in range(len(solved_tasks[0:-1])):

        ax.add_patch(mpl.patches.FancyArrowPatch(grid[task_names.index(solved_tasks[i])], grid[task_names.index(solved_tasks[i+1])]))

    ax.add_patch(mpl.patches.FancyArrowPatch(grid[task_names.index(solved_tasks[-1])], grid[task_names.index(solved_tasks[-1])]+[0,-0.6]))
    label(grid[task_names.index(solved_tasks[-1])]+[0,-0.7], f'{cp} баллов')

    plt.axis('equal')
    plt.axis('off')
    plt.show()

def task_solver(dbt, task_state, n_stages):
    
    st_level1 = 5*n_stages-3
    st_level2 = 3*n_stages-3
    
    past = '0'
    now = '12'
    cp = 0
    ck = 0

    solved_tasks = []
    correct_tasks = []

    while now != '1':

        solved_tasks.append(now[0:2])

        ck += 1

        task_id = task_state[now[0:2]]['id']

        question = dbt[dbt['ID'] == task_id].iloc[0]['Условие']
        solution = dbt[dbt['ID'] == task_id].iloc[0]['Решение']

        show_markdown('Тема: {} \n. Уровень сложности {}'.format(task_state[now[0:2]]['theme'],now[1]))

        show_html(question)

        ans = input('Ответ: ')

        if ans == 'p':
            show_markdown('Ответ правильный!')
            correct_tasks.append(now[0:2])
            if now[1] == '1':
                cp += 6
            elif now[1] == '2':
                cp +=3
            elif now[1] == '3':
                cp +=1
        elif ans == 'm':
            show_markdown('Ответ неверный. Вот решение: ')
            show_html(solution)

        else:
            show_markdown('Введён неверный символ.')

        now = now+ans
        cur = rules_dict.get((past,now))
        past = now
        now = cur

    show_markdown_big('Твой результат: {}'.format(cp))
    if cp > st_level1:
        show_markdown_big('Это великолепный результат!')
    elif cp< st_level2:
        show_markdown_big('Пожалуй тебе стоит поработать вот над этим номером КИМ.')
    else:
        show_markdown_big('Не плохо! Но уверен, ты можешь лучше!')

    return solved_tasks,correct_tasks,cp

На рисунке представлен график функции f(x)=x2a+bx+c, где числа а, b, с – целые. Определите значение дискриминанта уравнения f(x) = 0.

Ответ неверный. Вот решение:

Уравнение параболы по графику может быть определено через выражение y = k(x - xв)² + yв, где (xв; yв) – координаты вершины параболы, k – коэффициент, влияющий на растяжение/сжатие графика этой параболы.

По рисунку находим, что вершина параболы имеет координаты (-4; -9) и коэффициент k=12, тогда уравнение параболы имеет вид: f(x)=(x+4)22-9=x2+8x+162-9=x22+4x-1.

Тогда a = 2; b = 4; c = -1.

Найдем дискриминант уравнения x22+4x-1=0:

D = 16 - 4 · (-1) · 0,5 = 16 + 2 = 18.

Получим f(x)=5|x|-15.

Найдем f(12):
